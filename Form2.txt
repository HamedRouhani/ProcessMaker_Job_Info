      break;

if (JRT_Condition == 1 && PT_Condition == 1) {
    
    $("#submit0000000001").show();
  
  
  if (JRC == 99 ) {

        break;
        
      case (C2 > 8 && C2 < 99):   
        PC = '00' + (parseInt(C2) + 1);
        break;
    
      case (C2 > 98 && C2 < 999):   
        PC = '0' + (parseInt(C2) + 1);
        break;
    
      case (C2 > 998):   
        PC = (parseInt(C2) + 1);
        break;    
    
    }
  }
  
  $("#Job_Ranks_Code").setValue(JRC);
  $("#Post_Code").setValue(PC);
  
  switch(true) {
      
    case (EI_ID == "N"):
      $("#ORG_Job_Code").setValue(OC+"-"+PC+JRC);
      $("#ORG_Job_Title").setValue("”«“„«‰ : "+ON+" || " +"ÕÊ“Â : "+PT+" || "+"— »Â : "+JRT);
      $("#Job_Code").setValue(PC+JRC);

      $("#ORG_Job_Title").setValue("‰Ê⁄ ›⁄«·Ì  œ— ”«“„«‰ Œ«—ÃÌ : "+EON+" || " +" Œ’’ : "+PT+" || "+"œÌ”ÌÅ·Ì‰ : "+JRT);
